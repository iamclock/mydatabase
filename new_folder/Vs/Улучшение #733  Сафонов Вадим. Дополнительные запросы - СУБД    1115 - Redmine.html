SELECT fs.first_name, fs.surname, fs.patronymic
FROM
(
    SELECT f.id, f.first_name, f.surname, f.patronymic
    FROM freelancer f
    INNER JOIN skill s ON s.id_freelancer = f.id
    GROUP BY f.id HAVING COUNT(s.name) = ( SELECT COUNT(*) FROM skill s )
) AS fs
WHERE NOT EXISTS
(
    SELECT *
    FROM sending_message
    WHERE id_freelancer = fs.id
);

SELECT res.login, IF(@rowpos/2 < res.row_position, "активный спамер", "умеренный спамер") AS status
FROM
(
    SELECT @rowpos:=@rowpos+1 AS row_position, ems.login
    FROM
    (
    SELECT e.login, COUNT(sm.id) AS ctr
    FROM employer e    
    INNER JOIN sending_message sm ON sm.id_employer = e.id
    GROUP BY e.id ORDER BY COUNT(sm.id)
    ) AS ems, (SELECT @rowpos:=0) AS rp
) AS res;

SELECT SUM(t3.counters)
FROM
(
    SELECT t1.counters
    FROM
    (
    SELECT COUNT(*) AS counters
