Вывод: Капитан, Наибольшая сумма баллов

13. Для каждой команды известна дата её создания. Можно посчитать сколько других команд было создано раньше чем рассматриваемая команда. Для каждой команды нужно найти сколько других команд было создано раньше. Если это количество превышает 75% от общего числа команд, то рассматриваемой команде присвоить статус "новички". Если от 25% до 75%, то присвоить статус "основной состав". Если менее 25%, то присвоить статус "старожилы"
Вывод: Название команды, Капитан команды, Найденный статус.
14. Для каждого района найти всех людей, которые были там чаще остальных.
Вывод: Название района, ФИО человека, Количество раз, которое этот человек был в этом районе
15. Найти людей, для которых выполнено по крайней мере 4 условия из приведённых 5-ти
а) мужчина старше 20 лет
б) побывал в каком-либо походе в Китае
в) участвовал в какой-либо команде, занимавшей призовые места (т.е. места с 3-го по 1-ое)

SELECT head, max(sum_of_points)
FROM (
      SELECT f.place, head, SUM(r.points) AS sum_of_points
      FROM command c INNER JOIN competition co INNER JOIN fest f INNER JOIN result r
           ON co.command_id = c.id
              AND co.fest_id = f.id
              AND co.result_id = r.id
      GROUP BY head, f.place
     ) subq
GROUP BY head;

SELECT name, sport_grade, age FROM
(SELECT name, sport_grade, year(now())-year(birthday) as age
FROM
person p INNER JOIN was w INNER JOIN area a
ON w.person_id = p.id
   AND w.area_id = a.id
GROUP BY name
HAVING count(distinct a.title)=(select count(distinct id) from area)
) sub1
LEFT JOIN
(SELECT name as n
FROM
person p INNER JOIN participate pr INNER JOIN command c INNER JOIN competition co
ON pr.person_id = p.id
   AND pr.command_id = c.id
   AND co.command_id = c.id
) sub2 ON sub1.name = sub2.n
WHERE sub2.n IS NULL;

SELECT s5.title, s5.head, if(ct>(@alls:=(SELECT count(distinct id) FROM command))*0.75, "Новички", if(ct>=@alls*0.25, "Основной состав", "Старожилы")) AS status FROM
(   
        SELECT s2.title, s2.head, count(distinct s1.id) as ct FROM
            (SELECT id, head, title, creation_time FROM command) AS s1
            INNER JOIN 
            (SELECT head, title, creation_time FROM command) AS s2
            ON s1.creation_time < s2.creation_time
            GROUP BY s2.creation_time
        UNION
        SELECT * FROM (SELECT title, head FROM command WHERE creation_time=(SELECT MIN(creation_time) FROM command)) s1 INNER JOIN (SELECT 0 as ct) s2
) s5;

        WHERE p.person_id=SUBP.id and r.place<4
        ),1,0

        (
        SELECT w.person_id, count(distinct a.id) AS h_count
        FROM was w INNER JOIN area a ON w.area_id=a.id
        GROUP BY person_id
        HAVING person_id=SUBP.id AND h_count>10
        ),1,0
       )

    +if(EXISTS
        (
        SELECT g.person_id, sum(datediff(begin, end)) as days, h.id
        FROM goes g INNER JOIN hike h INNER JOIN area a
        ON g.hike_id = h.id
           AND h.area_id=a.id

SELECT CONCAT("Поход, начавшийся ",s1.begin," в район ",s1.title,", находящийся в стране ",s1.country," проходил под руководством ",s1.head,". В походе приняли участие ",s1.pc," человек из них ",FLOOR(s1.fc)," женщины и ",s1.mc," мужчины. В походе учавствовал фотограф, принимавший участие ещё в ",IF(s1.s2 is NULL,0,s1.s2)," походах. Закончился поход ",s1.end,".") as description
FROM
(
    (
    SELECT h.begin, h.end, a.title, a.country, h.head, h.photographer,
           @c:=count(distinct p.id) as pc, @male:=sum(p.sex) as mc, @c-@male as fc,
           (
            SELECT count(distinct gs.hike_id)
            FROM
            person pr INNER JOIN goes gs
            ON gs.person_id=pr.id
            GROUP BY pr.name
            HAVING pr.name=h.photographer
           ) AS s2
    FROM 
    person p INNER JOIN goes g INNER JOIN hike h INNER JOIN area a
    ON g.person_id = p.id
       AND g.hike_id = h.id
       AND h.area_id = a.id
    GROUP BY h.id
    ) AS s1
);

SELECT s1.c1t, s1.c1h, s1.c1ct, cm.title as c2t, cm.head as c2h, cm.creation_time as c2ct
FROM
    (SELECT cm.title as c1t, cm.head as c1h, cm.creation_time as c1ct, cm.id as c1, fest_id
    FROM command cm INNER JOIN competition cp
    ON cp.command_id = cm.id
    ) AS s1
    INNER JOIN
    (
    SELECT command_id as c2, fest_id
    FROM
    competition c
    ) AS s2
    INNER JOIN
    command cm
    ON s2.fest_id = s1.fest_id
       AND s1.c1 < s2.c2
       AND cm.id=s2.c2
GROUP BY c1t,c2t;

SELECT grup, count(distinct head) as captains_count
FROM (
      SELECT head,
      if(sp2>=bottom and sp2<b1,concat("[",bottom,";",b1,")"),if(sp2>=b1 and sp2<b2, concat("[",b1,";",b2,")"), concat("[",b2,";",top,")"))) as grup
      FROM (
            SELECT @mn:=min(sp) as bottom, @mx:=max(sp) as top, FLOOR(@mn+@step:=FLOOR((@mx-@mn)/3)) as b1, FLOOR(@mx-@step) as b2
            FROM (
                  SELECT head, max(sum_of_points) as sp
                  FROM (
                        SELECT f.place, head, SUM(r.points) AS sum_of_points
                        FROM command c INNER JOIN competition co INNER JOIN fest f INNER JOIN result r
                        ON co.command_id = c.id
                        AND co.fest_id = f.id
                        AND co.result_id = r.id
                  GROUP BY head, f.place
                  ) subq
            GROUP BY head
            ) AS s1
      ) AS s2,
     (
      SELECT head, max(sum_of_points) as sp2
      FROM (
            SELECT f.place, head, SUM(r.points) AS sum_of_points
            FROM command c INNER JOIN competition co INNER JOIN fest f INNER JOIN result r
            ON co.command_id = c.id
            AND co.fest_id = f.id
            AND co.result_id = r.id
            GROUP BY head, f.place
          ) subq
      GROUP BY head
    ) AS s3
) AS s4
GROUP by grup;
